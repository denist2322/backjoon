n = int(input())

result = ""

# 1. 함수정의


def star(x, y, n):
    global result
    if(x // n) % 3 == 1 and (y // n) % 3 == 1:
        # x, y를 현재 n값에 나누고 그 값의 3의 나머지가 1일 때
        result += " "
    elif n == 1:
        # 위 조건에 만족하지 않고 n이 1/3 까지 오면 별 출력
        result += "*"
    else:
        star(x, y, n//3)
        # n을 3으로 나누고, 함수 재호출


# 2. 계산 시작
for x in range(n):
    for y in range(n):
        star(x, y, n)
    result += "\n"

print(result)
